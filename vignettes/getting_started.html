Getting Started with spectreasy

spectreasy: Spectral Unmixing Quality Control

This vignette runs a small end-to-end spectreasy workflow with synthetic data. The code is evaluated during vignette builds, so it demonstrates actual package usage rather than static pseudocode.

suppressPackageStartupMessages(library(spectreasy))

Installation

For the released Bioconductor version, install with:

BiocManager::install("spectreasy")

If you need the current development version from GitHub, install with:

remotes::install_github("pkheisig/spectreasy")

remotes is used here rather than devtools because only GitHub installation is needed.

2. Build a toy flowSet in memory

A useful Bioconductor-friendly workflow is to keep multiple samples in memory as a flowSet. The helper below simulates three single-stain-like samples.

simulate_sample <- function(sample_name, dominant_marker, M, n_cells = 400) {
  markers <- rownames(M)
  marker_signal <- matrix(rexp(n_cells * length(markers), rate = 8), ncol = length(markers))
  colnames(marker_signal) <- markers
  marker_signal[, dominant_marker] <- rexp(n_cells, rate = 0.6) + 2

  raw_signal <- marker_signal %*% M + matrix(rnorm(n_cells * ncol(M), sd = 0.03), ncol = ncol(M))
  exprs_mat <- cbind(
    raw_signal,
    "Time" = seq_len(n_cells),
    "FSC-A" = rnorm(n_cells, mean = 90000, sd = 7000),
    "SSC-A" = rnorm(n_cells, mean = 45000, sd = 5000)
  )
  colnames(exprs_mat)[seq_len(ncol(M))] <- colnames(M)

  ff <- flowCore::flowFrame(exprs_mat)
  pd <- flowCore::pData(flowCore::parameters(ff))
  if (all(c("name", "desc") %in% colnames(pd))) {
    pd$desc <- pd$name
    flowCore::parameters(ff) <- methods::new("AnnotatedDataFrame", data = pd)
  }
  ff
}

toy_fs <- flowCore::flowSet(list(
  FITC_sample = simulate_sample("FITC_sample", "FITC", M_demo),
  PE_sample = simulate_sample("PE_sample", "PE", M_demo),
  APC_sample = simulate_sample("APC_sample", "APC", M_demo)
))

spectreasy also supports SingleCellExperiment, which fits more naturally into downstream Bioconductor single-cell workflows.

if (requireNamespace("SingleCellExperiment", quietly = TRUE) &&
    requireNamespace("SummarizedExperiment", quietly = TRUE) &&
    requireNamespace("S4Vectors", quietly = TRUE)) {
  toy_exprs <- do.call(cbind, lapply(seq_along(toy_fs), function(i) {
    t(flowCore::exprs(toy_fs[[i]]))
  }))
  toy_coldata <- data.frame(
    sample_id = rep(flowCore::sampleNames(toy_fs), each = nrow(flowCore::exprs(toy_fs[[1]]))),
    row.names = paste0("cell_", seq_len(ncol(toy_exprs))),
    stringsAsFactors = FALSE
  )
  colnames(toy_exprs) <- rownames(toy_coldata)

  toy_sce <- suppressWarnings(
    SingleCellExperiment::SingleCellExperiment(
      assays = list(counts = toy_exprs),
      colData = S4Vectors::DataFrame(toy_coldata)
    )
  )

  toy_unmixed_sce <- suppressWarnings(
    unmix_samples(
      sample_dir = toy_sce,
      M = M_demo,
      method = "OLS",
      write_fcs = FALSE,
      return_type = "SingleCellExperiment"
    )
  )

  class(toy_unmixed_sce)
  SummarizedExperiment::assayNames(toy_unmixed_sce)
  SingleCellExperiment::altExpNames(toy_unmixed_sce)
}

4. Generate diagnostic plots without writing files

Most plotting helpers return ggplot objects and only save when you pass an explicit output_file.

plot_spectra(M_demo, output_file = NULL)

plot_ssm(calculate_ssm(M_demo), output_file = NULL)

For SCC-like scatter diagnostics, provide a sample-to-marker mapping.

sample_to_marker <- c(
  FITC_sample = "FITC",
  PE_sample = "PE",
  APC_sample = "APC"
)

plot_unmixing_scatter_matrix(
  unmixed_list = toy_unmixed,
  sample_to_marker = sample_to_marker,
  markers = rownames(M_demo),
  output_file = NULL,
  max_points_per_sample = 150,
  seed = 1
)

We can also summarize negative-population spread from the unmixed results.

toy_results_df <- do.call(rbind, lapply(toy_unmixed, function(x) x$data))
plot_nps(calculate_nps(toy_results_df), output_file = NULL)

Real-data workflow

For real projects, organize your data like this:

my_project/
├── scc/                          # Single-color control FCS files
│   ├── FITC_beads.fcs
│   ├── PE_beads.fcs
│   └── Unstained.fcs             # Autofluorescence control
├── samples/                      # Experimental FCS files
│   └── Sample1.fcs
└── fcs_mapping.csv               # Control file (optional)

Create or review the control file

create_control_file(
  input_folder = "scc",
  cytometer = "Aurora",
  output_file = "fcs_mapping.csv"
)

Generate the SCC review report

generate_scc_report(
  scc_dir = "scc",
  control_file = "fcs_mapping.csv",
  cytometer = "Aurora",
  output_file = file.path("spectreasy_outputs", "SCC_QC_Report.pdf")
)

Build the reference matrix

control_df <- read.csv("fcs_mapping.csv", stringsAsFactors = FALSE, check.names = FALSE)

M <- build_reference_matrix(
  input_folder = "scc",
  output_folder = "gating_plots",
  save_qc_plots = TRUE,
  control_df = control_df,
  default_sample_type = "beads"
)

Run control-stage unmixing

ctrl <- autounmix_controls(
  scc_dir = "scc",
  control_file = "fcs_mapping.csv",
  cytometer = "Aurora",
  output_dir = "spectreasy_outputs/autounmix_controls"
)

Launch the interactive GUI

launch_gui(
  matrix_dir = "spectreasy_outputs/autounmix_controls",
  samples_dir = "samples"
)

Unmix experimental samples

unmixed <- unmix_samples(
  sample_dir = "samples",
  M = M,
  method = "WLS",
  output_dir = "spectreasy_outputs/unmix_samples",
  write_fcs = TRUE
)
